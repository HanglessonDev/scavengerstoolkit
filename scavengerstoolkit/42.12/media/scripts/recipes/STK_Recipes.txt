    /* R001 - Schoolbag (Large) → 2x Basic Straps + 1x Basic Fabric */
    craftRecipe DisassembleSchoolBag
    {
        allowbatchcraft = true,
        category        = Tailoring,
        tags            = AnySurfaceCraft,
        time            = 100,
        xpaward         = Tailoring:2,
        timedaction     = RipClothing,
        inputs
        {
            item 1 tags[Scissors] mode:keep flags[SharpnessCheck;IsNotDull],
            item 1 tags[Schoolbag] flags[IsNotWorn],
        }
        outputs
        {
            item 1 STK.BackpackFabricBasic,
            item 2 STK.BackpackStrapsBasic,
        }
    }
